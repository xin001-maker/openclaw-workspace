## 性能评估指标

### 核心指标

| 指标 | 含义 | 公式 | 优化方向 |
|------|------|------|----------|
| QPS | 每秒查询数 | 请求总数/时间 | 缓存、异步、扩容 |
| RT | 响应时间 | 请求处理耗时 | 优化代码、减少 IO |
| 吞吐量 | 系统处理能力 | QPS × 单次处理量 | 并行处理 |
| 可用性 | 系统可用程度 | 正常时间/总时间 | 冗余、降级 |
| 并发数 | 同时处理请求数 | 活跃线程数 | 线程池优化 |

### 性能计算公式

```
# 系统吞吐量
吞吐量 = QPS × 平均响应数据量

# 并发能力
最大并发数 = 线程数 × (1 + 等待时间/处理时间)

# 响应时间 (Little 定律)
平均响应时间 = 系统内请求数 / 吞吐量

# 服务器数量估算
服务器数量 = (总 QPS × 平均 RT) / 单服务器最大并发数

示例：
- 目标 QPS: 10,000
- 平均 RT: 50ms
- 单服务器最大并发：500
- 需要服务器：(10000 × 0.05) / 500 = 1 台 (理论值)
- 实际考虑冗余：3-5 台
```

### 本地缓存实现

```java
// Caffeine 本地缓存
import com.github.benmanes.caffeine.cache.Cache;
import com.github.benmanes.caffeine.cache.Caffeine;

public class LocalCache {
    
    private final Cache<String, Object> cache = Caffeine.newBuilder()
        .maximumSize(10_000)  // 最大 1 万条
        .expireAfterWrite(10, TimeUnit.MINUTES)  // 10 分钟过期
        .recordStats()
        .build();
    
    public Object get(String key) {
        return cache.getIfPresent(key);
    }
    
    public void put(String key, Object value) {
        cache.put(key, value);
    }
    
    public CacheStats stats() {
        return cache.stats();
    }
}

// 使用示例
LocalCache cache = new LocalCache();

    user = userService.getById(123);
    cache.put("user:123", user);
```

### 分布式缓存实现

```python
# Redis 缓存封装
import redis
import json
from typing import Optional, Any

class RedisCache:
    def __init__(self, host='localhost', port=6379, db=0):
        self.redis = redis.Redis(
            host=host,
            port=port,
            db=db,
            decode_responses=True,
            socket_connect_timeout=5,
            socket_timeout=5
        )
    
    def get(self, key: str) -> Optional[Any]:
        """获取缓存"""
        data = self.redis.get(key)
        if data:
            return json.loads(data)
        return None
    
    def set(self, key: str, value: Any, expire: int = 300) -> bool:
        """设置缓存"""
        try:
            self.redis.setex(
                key,
                expire,
                json.dumps(value, ensure_ascii=False)
            )
            return True
        except Exception as e:
            print(f"Redis set error: {e}")
            return False
    
    def delete(self, key: str) -> bool:
        """删除缓存"""
        return bool(self.redis.delete(key))
    
    def exists(self, key: str) -> bool:
        """检查是否存在"""
        return bool(self.redis.exists(key))
    
    def incr(self, key: str, expire: int = 300) -> int:
        """原子自增"""
        count = self.redis.incr(key)
        if count == 1:
            self.redis.expire(key, expire)
        return count
    
    # 防缓存穿透
    def get_with_bloom(self, key: str, bloom_filter) -> Optional[Any]:
        """带布隆过滤器的查询"""
        # 先检查布隆过滤器
        if not bloom_filter.contains(key):
            return None  # 肯定不存在
        
        # 查询缓存
        data = self.get(key)
        if data is not None:
            return data
        
        # 查询数据库后缓存空值 (防穿透)
        # db_value = db.query(...)
        # if db_value is None:
        #     self.set(key, '__NULL__', expire=60)
        #     return None
        
        return data

# 缓存装饰器
def cache_result(key_prefix: str, expire: int = 300):
    def decorator(func):
        def wrapper(*args, **kwargs):
            cache_key = f"{key_prefix}:{args[0] if args else ''}"
            
            # 尝试从缓存获取
            cached = redis_cache.get(cache_key)
            if cached is not None:
                return cached
            
            # 执行函数
            result = func(*args, **kwargs)
            
            # 写入缓存
            if result is not None:
                redis_cache.set(cache_key, result, expire)
            
            return result
        return wrapper
    return decorator

# 使用示例
@cache_result("user:profile", expire=600)
def get_user_profile(user_id: int):
    return db.query("SELECT * FROM users WHERE id = ?", user_id)
```

### 缓存更新策略

```python
# 策略 1: Cache Aside (旁路缓存) - 推荐
def get_user(user_id):
    # 读：先读缓存，未命中读 DB
    user = cache.get(f"user:{user_id}")
    if user is None:
        user = db.get_user(user_id)
        if user:
            cache.set(f"user:{user_id}", user, expire=300)
    return user

def update_user(user_id, data):
    # 写：先更新 DB，再删除缓存
    db.update_user(user_id, data)
    cache.delete(f"user:{user_id}")

# 策略 2: Read/Write Through
def get_user(user_id):
    # 缓存负责加载数据
    return cache.get_through(f"user:{user_id}", 
                            lambda: db.get_user(user_id))

# 策略 3: Write Behind (异步写回)
def update_user(user_id, data):
    # 只更新缓存，异步写回 DB
    cache.set(f"user:{user_id}", data)
    mq.send("user_update", {"user_id": user_id, "data": data})

# 策略对比
┌─────────────────────────────────────────────────────┐
```

```python
# 分库分表策略
class DatabaseSharding:
    def __init__(self):
        # 16 个库，每个库 64 张表
        self.db_count = 16
        self.table_count = 64
    
    def get_db_index(self, user_id: int) -> int:
        """根据用户 ID 计算库索引"""
        return user_id % self.db_count
    
    def get_table_index(self, user_id: int) -> int:
        """根据用户 ID 计算表索引"""
        return user_id % self.table_count
    
    def get_connection(self, user_id: int):
        """获取数据库连接"""
        db_idx = self.get_db_index(user_id)
        return connections[f"db_{db_idx}"]
```

### 读写分离

```python
# 读写分离配置
from sqlalchemy import create_engine
from sqlalchemy.orm import sessionmaker

# 主库 (写)
master_engine = create_engine(
    "mysql+pymysql://user:pass@master:3306/app"
)

# 从库 (读) - 多个
slave_engines = [
    create_engine("mysql+pymysql://user:pass@slave1:3306/app"),
    create_engine("mysql+pymysql://user:pass@slave2:3306/app"),
    create_engine("mysql+pymysql://user:pass@slave3:3306/app"),
]

# 路由选择
class RoutingSession:
    def __init__(self):
        self.slave_index = 0
    
    def get_engine(self, is_write: bool):
        if is_write:
            return master_engine
        else:
            # 轮询从库
            engine = slave_engines[self.slave_index]
            self.slave_index = (self.slave_index + 1) % len(slave_engines)
            return engine
    
    def query(self, sql, params=None):
        engine = self.get_engine(is_write=False)
        with engine.connect() as conn:
            return conn.execute(sql, params or [])
    
    def execute(self, sql, params=None):
        engine = self.get_engine(is_write=True)
        with engine.connect() as conn:
            return conn.execute(sql, params or [])

# 使用示例
session = RoutingSession()

# 读操作 → 从库
users = session.query("SELECT * FROM users WHERE id = ?", [1])

# 写操作 → 主库
session.execute("UPDATE users SET name = ? WHERE id = ?", ["John", 1])
```

### SQL 优化

```sql
-- ❌ 慢查询示例
SELECT * FROM orders o
JOIN users u ON o.user_id = u.id
JOIN products p ON o.product_id = p.id
WHERE YEAR(o.created_at) = 2024
ORDER BY o.created_at;

-- ✅ 优化后
SELECT o.id, o.amount, u.name, p.title
FROM orders o
JOIN users u ON o.user_id = u.id
JOIN products p ON o.product_id = p.id
WHERE o.created_at >= '2024-01-01' AND o.created_at < '2025-01-01'
ORDER BY o.created_at DESC
LIMIT 100;

-- 优化要点：
-- 1. 只 SELECT 需要的字段
-- 2. 避免函数操作索引字段
-- 3. 添加 LIMIT 限制
-- 4. 使用覆盖索引
```

### 限流算法

```python
# 1. 固定窗口计数器
class FixedWindowRateLimiter:
    def __init__(self, limit: int, window_seconds: int):
        self.limit = limit
        self.window = window_seconds
        self.counts = {}  # {key: (count, window_start)}
    
    def allow(self, key: str) -> bool:
        now = time.time()
        if key not in self.counts:
            self.counts[key] = (1, now)
            return True
        
        count, window_start = self.counts[key]
        if now - window_start >= self.window:
            # 新窗口
            self.counts[key] = (1, now)
            return True
        
        if count >= self.limit:
            return False
        
        self.counts[key] = (count + 1, window_start)
        return True

# 2. 滑动窗口
class SlidingWindowRateLimiter:
    def __init__(self, limit: int, window_seconds: int):
        self.limit = limit
        self.window = window_seconds
        self.requests = defaultdict(list)  # {key: [timestamps]}
    
    def allow(self, key: str) -> bool:
        now = time.time()
        cutoff = now - self.window
        
        # 清理过期请求
        self.requests[key] = [
            ts for ts in self.requests[key] if ts > cutoff
        ]
        
        if len(self.requests[key]) >= self.limit:
            return False
        
        self.requests[key].append(now)
        return True

# 3. 令牌桶 (Redis 实现)
class TokenBucketRateLimiter:
    def __init__(self, redis_client, key: str, capacity: int, refill_rate: float):
        self.redis = redis_client
        self.key = f"rate_limit:{key}"
        self.capacity = capacity
        self.refill_rate = refill_rate  # 每秒补充的令牌数
    
    def allow(self) -> bool:
        now = time.time()
        pipe = self.redis.pipeline()
        
        # 获取当前令牌数和上次更新时间
        pipe.hget(self.key, 'tokens')
        pipe.hget(self.key, 'updated_at')
        results = pipe.execute()
        
        tokens = float(results[0]) if results[0] else self.capacity
        last_update = float(results[1]) if results[1] else now
        
        # 补充令牌
        elapsed = now - last_update
        tokens = min(self.capacity, tokens + elapsed * self.refill_rate)
        
        if tokens < 1:
            return False
        
        # 消耗令牌
        tokens -= 1
        pipe = self.redis.pipeline()
        pipe.hset(self.key, mapping={'tokens': tokens, 'updated_at': now})
        pipe.expire(self.key, 3600)
        pipe.execute()
        
        return True

# 4. 漏桶
class LeakyBucketRateLimiter:
    def __init__(self, capacity: int, leak_rate: int):
        self.capacity = capacity  # 桶容量
        self.leak_rate = leak_rate  # 每秒漏出量
        self.buckets = {}  # {key: (water, last_leak)}
    
    def allow(self, key: str) -> bool:
        now = time.time()
        
        if key not in self.buckets:
            self.buckets[key] = (0, now)
        
        water, last_leak = self.buckets[key]
        
        # 漏水
        elapsed = now - last_leak
        water = max(0, water - elapsed * self.leak_rate)
        
        if water >= self.capacity:
            return False
        
        # 加水
        water += 1
        self.buckets[key] = (water, now)
        return True
```

```python
# 熔断器实现
class CircuitBreaker:
    def __init__(self, failure_threshold=5, recovery_timeout=30):
        self.failure_threshold = failure_threshold
        self.recovery_timeout = recovery_timeout
        self.failures = 0
        self.last_failure_time = None
        self.state = 'CLOSED'  # CLOSED, OPEN, HALF_OPEN
    
    def call(self, func, *args, **kwargs):
        if self.state == 'OPEN':
            if time.time() - self.last_failure_time > self.recovery_timeout:
                self.state = 'HALF_OPEN'
            else:
                raise CircuitBreakerOpenError()
        
        try:
            result = func(*args, **kwargs)
            if self.state == 'HALF_OPEN':
                self.state = 'CLOSED'
                self.failures = 0
            return result
        except Exception as e:
            self.failures += 1
            self.last_failure_time = time.time()
            
            if self.failures >= self.failure_threshold:
                self.state = 'OPEN'
            
            raise e

# 使用示例
breaker = CircuitBreaker(failure_threshold=5, recovery_timeout=30)

def get_user_with_fallback(user_id):
    try:
        return breaker.call(user_service.get_user, user_id)
    except CircuitBreakerOpenError:
        # 熔断，返回缓存
        return cache.get_user(user_id)
    except Exception:
        # 其他异常，降级处理
        return User.default()
```

```python
import random
import hashlib

class LoadBalancer:
    def __init__(self, services: list):
        self.services = services
        self.index = 0
    
    # 轮询
    def round_robin(self) -> str:
        service = self.services[self.index]
        self.index = (self.index + 1) % len(self.services)
        return service
    
    # 随机
    def random_select(self) -> str:
        return random.choice(self.services)
    
    # 一致性 Hash
    def consistent_hash(self, key: str) -> str:
        hash_value = int(hashlib.md5(key.encode()).hexdigest(), 16)
        index = hash_value % len(self.services)
        return self.services[index]
    
    # 加权轮询
    def weighted_round_robin(self, weights: dict) -> str:
        # weights: {'service1': 3, 'service2': 1}
        weighted_list = []
        for service, weight in weights.items():
            weighted_list.extend([service] * weight)
        return random.choice(weighted_list)
```

```python
# Redis 分布式锁
import redis
import time
import uuid

class RedisLock:
    def __init__(self, redis_client, lock_name: str, timeout: int = 10):
        self.redis = redis_client
        self.lock_name = f"lock:{lock_name}"
        self.timeout = timeout
        self.lock_id = str(uuid.uuid4())
    
    def acquire(self) -> bool:
        # SETNX + 过期时间 (原子操作)
        result = self.redis.set(
            self.lock_name,
            self.lock_id,
            nx=True,
            ex=self.timeout
        )
        return result is not None
    
    def release(self) -> bool:
        # Lua 脚本保证原子性
        lua_script = """
        if redis.call('get', KEYS[1]) == ARGV[1] then
            return redis.call('del', KEYS[1])
        else
            return 0
        end
        """
        result = self.redis.eval(lua_script, 1, self.lock_name, self.lock_id)
        return result == 1
    
    def __enter__(self):
        while not self.acquire():
            time.sleep(0.1)
        return self
    
    def __exit__(self, exc_type, exc_val, exc_tb):
        self.release()

# 使用示例
with RedisLock(redis_client, "order:123", timeout=10):
    # 临界区代码
```

```python
class FlashSaleService:
    def __init__(self):
        self.redis = redis.Redis()
        self.mq = KafkaProducer()
    
    def flash_sale(self, user_id: int, product_id: int):
        # 1. 检查是否重复购买
        if self.redis.sismember(f"flash:buyers:{product_id}", user_id):
            raise Exception("已购买过")
        
        # 2. Redis 原子扣减库存
        stock = self.redis.decr(f"flash:stock:{product_id}")
        if stock < 0:
            self.redis.incr(f"flash:stock:{product_id}")
            raise Exception("库存不足")
        
        # 3. 发送订单消息
        self.mq.send('flash-orders', {
            'user_id': user_id,
            'product_id': product_id,
            'timestamp': time.time()
        })
        
        # 4. 标记已购买
        self.redis.sadd(f"flash:buyers:{product_id}", user_id)
        
        return {'status': 'queued', 'message': '排队中'}
    
    def process_order(self, message):
        # 异步处理订单
        try:
            create_order_in_db(message)
            self.redis.incr(f"flash:sold:{message['product_id']}")

            self.redis.incr(f"flash:stock:{message['product_id']}")
```

```python
        if not long_url:
            # 缓存未命中，查 DB
            long_url = self._get_from_db(short_code)
            if long_url:
                self.redis.set(f"url:short:{short_code}", long_url, ex=86400*30)
        
        if long_url:
            # 异步记录访问
            self._record_access(short_code)
        
        return long_url
    
    def _to_base62(self, num: int) -> str:
        if num == 0:
            return self.BASE62[0]
        
        result = []
        while num > 0:
            result.append(self.BASE62[num % 62])
            num //= 62
        
        return ''.join(reversed(result))
```
